import java.util.LinkedList;

// 定义二叉树节点
class TreeNode {
    int val;
    TreeNode left;
    TreeNode right;

    TreeNode(int x) {
        val = x;
    }
}

public class SortedLinkedListToBST {

    private LinkedList<Integer> list; // 存储链表中的元素

    // 主方法：从排序链表构建平衡二叉搜索树
    public TreeNode sortedListToBST(LinkedList<Integer> linkedList) {
        this.list = linkedList; // 记录链表
        return sortedListToBSTHelper(0, list.size() - 1); // 递归构建 BST
    }

    // 递归方法：构建平衡二叉搜索树
    private TreeNode sortedListToBSTHelper(int left, int right) {
        if (left > right) {
            return null; // 基础条件：当左边界大于右边界时，返回 null
        }

        int mid = left + (right - left) / 2; // 计算中间索引

        // 递归构建左子树
        TreeNode leftChild = sortedListToBSTHelper(left, mid - 1);

        // 使用链表中当前中间元素的值创建节点
        TreeNode node = new TreeNode(list.get(mid));

        node.left = leftChild; // 连接左子树

        // 递归构建右子树
        node.right = sortedListToBSTHelper(mid + 1, right);

        return node; // 返回当前节点
    }

    // 辅助方法：用于打印树的前序遍历
    public void preOrder(TreeNode node) {
        if (node != null) {
            System.out.print(node.val + " "); // 访问根节点
            preOrder(node.left); // 访问左子树
            preOrder(node.right); // 访问右子树
        }
    }

    // 测试代码
    public static void main(String[] args) {
        // 创建一个示例排序链表: 1 -> 2 -> 3 -> 4 -> 5 -> 6
        LinkedList<Integer> linkedList = new LinkedList<>();
        linkedList.add(1);
        linkedList.add(2);
        linkedList.add(3);
        linkedList.add(4);
        linkedList.add(5);
        linkedList.add(6);

        SortedLinkedListToBST converter = new SortedLinkedListToBST();
        TreeNode root = converter.sortedListToBST(linkedList); // 构建平衡二叉搜索树

        // 打印树的前序遍历
        System.out.println("Preorder traversal of the constructed BST:");
        converter.preOrder(root);
    }
}
